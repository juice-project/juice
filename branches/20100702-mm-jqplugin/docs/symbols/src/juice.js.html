 44 /**
 45  * Create and store meta reference from DOM elements or element attributes using jQuery selectors.
 46  * @see #setMeta
 47  * @param {String} id The id of meta definition to create
 48  * @param {String} selector JQuery selection string for element within page
 49  * @param {String} [attribute] Attribute name if attribute values are wanted
 50  * @param {Function} [filter] Function used to process retrieved data before storage
 51  */
 52 _Juice.prototype.findMeta = function(id, selector, attribute, filter){
 53 	
 54 	if ( $jq.isFunction(attribute) ) {
 55 		filter = attribute;
 56 		attribute = null;
 57     }
 58 	var i = 0;
 59 	var values = [];
 60 	$jq(selector).each(function(){
 61 		var val;
 62 		if(attribute){
 63 			val = $jq(this).attr(attribute);
 64 		}else{
 65 			if ($jq(this).size() > 0) {
 66 				//Only want the text of this node - not child nodes 
 67 				// TODO: better save a jQuery object and filter filter nodes?
 68 				val = "";
 69 				var contents = $jq(this).contents();
 70 				contents.each(function(){if(this.nodeType == 3 ) val += this.nodeValue;});
 71                 // TODO: should we normalize space here?
 72 			}
 73 		}
 74 		if($jq.isFunction(filter) && val !== undefined ){
 75 			val = filter(val,id); // TODO: why was THIS given as second parameter to filter?
 76 		}
 77 		if (val !== undefined) {
 78 			values[i++] = val;
 79 		}
 80 	});
 81 	if(values.length > 0){
 82 		this.setMeta(id,values);
 83 	}
 84 	
 85 }

398 _Juice.prototype._loadFile = function (file,type,evnt){
399 	
400 	        if(type == "js"){
401 	        		var This = this;
402 	           		var head = document.getElementsByTagName('head')[0]; 
403 	            	var ins = document.createElement('script'); 
404 	            	evnt = evnt || $jq.noop;
405 	            	ins.type = 'text/javascript'; 
406 	            	ins.src = file; 
407 	            	this.JsLoadFlags[this.JsLoadFlags.length] = {'name':file, 'loaded':false};
408 	            	ins.onreadystatechange = function () {
409 	            	    if (ins.readyState == 'loaded' || ins.readyState == 'complete') {        
410 	            	        evnt();      
411 	            	        This.jsOnLoadEvent(file);
412 	            	    }
413 	            	}
414 	            	ins.onload = function () {    
415 	            	   evnt();       
416 	            	   This.jsOnLoadEvent(file);
417 	            	}
418 	            	head.insertBefore(ins, head.firstChild );
419 	            	
420 	        }else if(type == "css"){
421 	            $('head').prepend('<link type="text/css" rel="stylesheet" href="'+file+'" />');

428 _Juice.prototype._absoluteUri = function(file, pathPrefix){
429 	if(this._strBeginsWith(file,"http://") || this._strBeginsWith(file,"https://") || this._strBeginsWith(window.location.protocol,"file:") ){
430 		return file;
431 	}
432 	if(this._strBeginsWith(file,"/")){
433 		if(!pathPrefix){
434 			pathPrefix = "";
435 		}else{
436 			pathPrefix = "/" + pathPrefix;
437 		}
438 		return window.location.protocol + "//" + window.location.host + pathPrefix + file;
439 	}else{
440 		if(!pathPrefix){
441 			pathPrefix = "";
442 		}else{
443 			pathPrefix = pathPrefix + "/";
444 		}
445 		var ret =  window.location.protocol + "//" + window.location.host;
446 		var path = document.location.pathname;
447 		if(this._strEndsWith(path,"/")){
448 			path = path.substr(0,path.length -1);
449 		}
450 		
451 		var locationBits = path.split('/');
452 		for(var i = 0; i < locationBits.length - 1;i++){
453 			ret += locationBits[i] + "/";
454 		}
455 		return ret + pathPrefix + file;
456 	}
457 }

640 _Juice.prototype.stringToArray = function(str,extras)

706 _Juice.prototype.setCookie = function(name,value,minutes,hours,days,path,domain,secure ){
707 	var cookie_string = name + "=" + escape ( value );
708 
709 	if(minutes){
710 		if(!hours){
711 			hours = 0;
712 		}
713 		if(!days){
714 			days = 0;
715 		}
716 		var date = new Date();
717 		date.setTime(date.getTime()+((((days*24*60*60) + (hours*60*60) + (minutes*60))*1000)));
718 		var expires = "; expires="+date.toGMTString();
719 		cookie_string += "; expires=" + expires.toGMTString();
720 	}
721 
722 	if(path){
723 		cookie_string += "; path=" + escape ( path );
724 	}
725 
726 	if (domain){
727 		cookie_string += "; domain=" + escape ( domain );	
728 	}
729 
730 	if (secure){
731 		cookie_string += "; secure";	
732 	}
733 
734 	document.cookie = cookie_string;

776 /**
777  * Definition of insert in to document body
778  * InsertPoint could result in multiple instances of insert on a single page - this is supported
779  * methods such as show(), getInsertObject(), and remove() default to a zero position in any
780  * array of instances to simplify operation of a single insert instance.
781  * @namespace
782  * @constructor
783  * @param {String} container html to be inserted in to document
784  * @param {String} insertPoint location within document
785  * @param {String} insertType How to insert at insert point: before | after | append | prepend | replace
786  */
787 
788 function JuiceInsert(container,insertPoint,insertType){
789 	//html/dom to insert in to page
790 	this._container = container;
791 	//JQuery selection identifying insert point(s) in document
792 	this._insertPoint = insertPoint;
793 	//How to insert at insert point: before | after | append | prepend | replace
794 	this._insertType = insertType;
795 	//Shown flags
796 	this.shown = [];
797 	//JQuery/Dom elements created on insertion
798 	this.insertObjects = [];
799 	//Number of matching insert points
800 	this.inserts = $jq(this._insertPoint).length;
801 
802 	for(var i=0;i< this.inserts; i++){
803 		this.shown[i] = false;
804 		this.insertObjects[i] = null;
805 	}
806 }

809 JuiceInsert.prototype.showAll = function(){
810 	for(var i=0;i< this.inserts; i++){
811 		this.show(i);
812 	}
813 }

815 //show - If not shown, insert container in document
816 //arg: pos - which instance of this insert to show - optional - defaults to 0
817 
818 JuiceInsert.prototype.show = function(pos){
819 	pos = pos || 0;
820 	var This = this;
821 	$jq(this._insertPoint).each(function(i){
822 		if(i == pos && !This.shown[i]){
823 			var ins = jQuery(This._container);
824 			var target = jQuery(this);
825 			This.insertObjects[i] = ins;
826 			switch(This._insertType){
827 				case "after":
828 					target.after(ins);
829 					break;
830 				case "before":
831 					target.before(ins);
832 					break;
833 				case "prepend":
834 					target.prepend(ins);
835 					break;
836 				case "replace":
837 					target.replaceWith(ins);
838 					break;
839 				case "append":
840 				default:
841 					target.append(ins);
842 					break;
843 			}
844 			This.shown[i] = true;
845 		}		
846 	});
847 }
