163 _Juice.prototype._selectMeta = function(id, selector, attribute, filter){
164 	if ( $jq.isFunction(attribute) ) {
165 		filter = attribute;
166 		attribute = null;
167     }
168 	var i = 0;
169 	var values = [];
170 	$jq(selector).each(function(){
171 		var val;
172 		if(attribute){
173 			val = $jq(this).attr(attribute);
174 		}else{
175 			if ($jq(this).size() > 0) {
176 				//Only want the text of this node - not child nodes 
177 				// TODO: better save a jQuery object and filter filter nodes?
178 				val = "";
179 				var contents = $jq(this).contents();
180 				contents.each(function(){if(this.nodeType == 3 ) val += this.nodeValue;});
181                 // TODO: should we normalize space here?
182 			}
183 		}
184 		if(jQuery.isFunction(filter) && val !== undefined ){
185 			val = filter(val,id); // TODO: why was THIS given as second parameter to filter?
186 		}
187 		if (val !== undefined) {
188 			values[i++] = val;
189 		}
190 	});
191 	if(values.length > 0){
192 		return values;
193 	}
194 	return null;
195 }

559 //_loadFile - internl function to append file elements to document header
560 //arg: type - "function to call when loaded "css" | "js"
561 //arg: onLoadEvent - function to call when loaded - only relevant for type "js"
562 _Juice.prototype._loadFile = function (file,type,onLoadEvent){
563 	var This = this;
564 	var evnt = onLoadEvent;
565     var head = document.getElementsByTagName('head')[0]; 
566     var ins = null;
567 	if(type == "js"){
568 	 	ins = document.createElement('script'); 
569 	    ins.type = 'text/javascript'; 
570 	    ins.src = file; 
571 	}else if(type == "css"){
572 	 	ins = document.createElement('link'); 
573 	    ins.type = 'text/css'; 
574 	    ins.rel = 'stylesheet'; 		
575 	    ins.href = file; 		
576 	}
577 
578 	if(type == "js"){
579 		This.JsLoadFlags[this.JsLoadFlags.length] = new JsLoadFlag(file);
580 		
581 		ins.onreadystatechange = function () {
582 	        if (ins.readyState == 'loaded' || ins.readyState == 'complete') {
583 				if(evnt){
584 					evnt();
585 				}
586 	            This.jsOnLoadEvent(file);
587 	        }
588 	    }
589 	    ins.onload = function () {
590 			if(evnt){
591 				evnt();
592 			}
593 	       This.jsOnLoadEvent(file);
594 	    }
595 	}
596 	//Copy of jQuery fix for IE6 - insert at top of head 
597 	head.insertBefore( ins, head.firstChild );
598 }

600 _Juice.prototype._absoluteUri = function(file, pathPrefix){
601 
602 	if(this._strBeginsWith(file,"http://") || this._strBeginsWith(file,"https://")){
603 		return file;
604 	}
605 	if(this._strBeginsWith(file,"/")){
606 		if(!pathPrefix){
607 			pathPrefix = "";
608 		}else{
609 			pathPrefix = "/" + pathPrefix;
610 		}
611 		return window.location.protocol + "//" + window.location.host + pathPrefix + file;
612 	}else{
613 		if(!pathPrefix){
614 			pathPrefix = "";
615 		}else{
616 			pathPrefix = pathPrefix + "/";
617 		}
618 		var ret =  window.location.protocol + "//" + window.location.host;
619 		var path = document.location.pathname;
620 		if(this._strEndsWith(path,"/")){
621 			path = path.substr(0,path.length -1);
622 		}
623 		
624 		var locationBits = path.split('/');
625 		for(var i = 0; i < locationBits.length - 1;i++){
626 			ret += locationBits[i] + "/";
627 		}
628 		return ret + pathPrefix + file;
629 	}
630 }

632 //findJs - return true if script element already loaded in document
633 _Juice.prototype.findJs = function (file){
634 	var fileString = this.urlRoot(file);
635 	var scripts = document.getElementsByTagName('script');
636 	
637 	for(var i=0;i < scripts.length;i++){
638 		if(this.urlRoot(scripts[i].getAttribute("src")) == fileString){
639 			return true;
640 		}
641 	}
642 	return false;
643 }

645 //Remove everything after and including '#' and/or '?' from a string
646 _Juice.prototype.urlRoot = function(url){
647 	str = new String(url);
648 	if(str.indexOf('#') != -1){
649 		str = new String(str.substring(0,str.indexOf('#')));
650 	}
651 	if(str.indexOf('?') != -1){
652 		str = new String(str.substring(0,str.indexOf('?')));
653 	}
654 	return str.toString();
655 }

849 _Juice.prototype.stringToArray = function(str,extras)
850 // Assumes: str is a sequence of words, separated by whitespace
851 // Other seperator chars can be added from string extras
852 // Returns: an array containing the individual words
853 {
854 	var seps = " \t\n\r" + extras;
855 	var index;
856

859 
860 	while (str != null && str != "") {

863 	    index++;  
864 	  }

866 	    var item = "";
867 	    while (index < str.length && !juice.isSepChar(str.charAt(index),seps)) {
868 	      item += str.charAt(index);
869 	      index++;
870 	    }
871 
872 	    items[count++] = item;
873 	  }
874 	  str = str.substring(index+1, str.length);
875 	}
876   return items;
