156  * Identify data from DOM elements or element attributes using jQuery selectors.
157  * @param {String} id The id of meta definition to create
158  * @param {String} selector JQuery selection string for element within page

164 _Juice.prototype._selectMeta = function(id, selector, attribute, filter){
165 	if ( $jq.isFunction(attribute) ) {
166 		filter = attribute;
167 		attribute = null;
168     }
169 	var i = 0;
170 	var values = [];
171 	$jq(selector).each(function(){
172 		var val;
173 		if(attribute){
174 			val = $jq(this).attr(attribute);
175 		}else{
176 			if ($jq(this).size() > 0) {
177 				//Only want the text of this node - not child nodes 
178 				// TODO: better save a jQuery object and filter filter nodes?
179 				val = "";
180 				var contents = $jq(this).contents();
181 				contents.each(function(){if(this.nodeType == 3 ) val += this.nodeValue;});
182                 // TODO: should we normalize space here?
183 			}
184 		}
185 		if(jQuery.isFunction(filter) && val !== undefined ){
186 			val = filter(val,id); // TODO: why was THIS given as second parameter to filter?
187 		}
188 		if (val !== undefined) {
189 			values[i++] = val;
190 		}
191 	});
192 	if(values.length > 0){
193 		return values;
194 	}
195 	return null;
196 }

563 //_loadFile - internl function to append file elements to document header
564 //arg: type - "function to call when loaded "css" | "js"
565 //arg: onLoadEvent - function to call when loaded - only relevant for type "js"
566 _Juice.prototype._loadFile = function (file,type,onLoadEvent){
567 	var This = this;
568 	var evnt = onLoadEvent;
569     var head = document.getElementsByTagName('head')[0]; 
570     var ins = null;
571 	if(type == "js"){
572 	 	ins = document.createElement('script'); 
573 	    ins.type = 'text/javascript'; 
574 	    ins.src = file; 
575 	}else if(type == "css"){
576 	 	ins = document.createElement('link'); 
577 	    ins.type = 'text/css'; 
578 	    ins.rel = 'stylesheet'; 		
579 	    ins.href = file; 		
580 	}
581 
582 	if(type == "js"){
583 		This.JsLoadFlags[this.JsLoadFlags.length] = new JsLoadFlag(file);
584 		
585 		ins.onreadystatechange = function () {
586 	        if (ins.readyState == 'loaded' || ins.readyState == 'complete') {
587 				if(evnt){
588 					evnt();
589 				}
590 	            This.jsOnLoadEvent(file);
591 	        }
592 	    }
593 	    ins.onload = function () {
594 			if(evnt){
595 				evnt();
596 			}
597 	       This.jsOnLoadEvent(file);
598 	    }
599 	}
600 	//Copy of jQuery fix for IE6 - insert at top of head 
601 	head.insertBefore( ins, head.firstChild );
602 }

604 _Juice.prototype._absoluteUri = function(file, pathPrefix){
605 
606 	if(this._strBeginsWith(file,"http://") || this._strBeginsWith(file,"https://")){
607 		return file;
608 	}
609 	if(this._strBeginsWith(file,"/")){
610 		if(!pathPrefix){
611 			pathPrefix = "";
612 		}else{
613 			pathPrefix = "/" + pathPrefix;
614 		}
615 		return window.location.protocol + "//" + window.location.host + pathPrefix + file;
616 	}else{
617 		if(!pathPrefix){
618 			pathPrefix = "";
619 		}else{
620 			pathPrefix = pathPrefix + "/";
621 		}
622 		var ret =  window.location.protocol + "//" + window.location.host;
623 		var path = document.location.pathname;
624 		if(this._strEndsWith(path,"/")){
625 			path = path.substr(0,path.length -1);
626 		}
627 		
628 		var locationBits = path.split('/');
629 		for(var i = 0; i < locationBits.length - 1;i++){
630 			ret += locationBits[i] + "/";
631 		}
632 		return ret + pathPrefix + file;
633 	}
634 }

636 //findJs - return true if script element already loaded in document
637 _Juice.prototype.findJs = function (file){
638 	var fileString = this.urlRoot(file);
639 	var scripts = document.getElementsByTagName('script');
640 	
641 	for(var i=0;i < scripts.length;i++){
642 		if(this.urlRoot(scripts[i].getAttribute("src")) == fileString){
643 			return true;
644 		}
645 	}
646 	return false;
647 }

649 //Remove everything after and including '#' and/or '?' from a string
650 _Juice.prototype.urlRoot = function(url){
651 	str = new String(url);
652 	if(str.indexOf('#') != -1){
653 		str = new String(str.substring(0,str.indexOf('#')));
654 	}
655 	if(str.indexOf('?') != -1){
656 		str = new String(str.substring(0,str.indexOf('?')));
657 	}
658 	return str.toString();
659 }
